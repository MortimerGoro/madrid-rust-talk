Rust hygienic macro system

macro expansion happens in a distinct syntax context


macro_rules! log {
  ($msg:expr) => {{
	let state: i32 = get_log_state();
	if state > 0 {
	  println!("log({}): {}", state, $msg);
	}
  }};
}
